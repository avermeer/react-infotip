export const parseBorder = border => { const [, , width, style, , color] = /^(([1-9]\d*)px\s+)?(none|solid|dotted|dashed)(\s+(.+))?$/.exec(border) || [] return { width, style, color } } export const parseBoxShadow = boxShadow => { const [, , dx, , dy, color] = /^(([1-9]\d*)px\s+)(([1-9]\d*)px\s+)(.+)$/.exec(boxShadow) || [] return { dx, dy, color } }
